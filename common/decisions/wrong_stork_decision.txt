spawn_female_child = {
	picture = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"
	title = "Spawn Female Child"
    desc = "Spawns a female Child."
    selection_tooltip = "This will give you a female Child."
    confirm_text = "Ok"

	is_shown = {
		debug_only = yes
	}

	effect = {
		if = {
			limit = { is_female = yes }
			save_scope_as = mother
			scope:spouse = { save_scope_as = father }
		}
		else = {
			save_scope_as = father
			scope:spouse = { save_scope_as = mother }
		}

		create_character = {
			age = { 1 15 }
			gender_female_chance = 100
			trait = wild_oat
			mother = scope:mother
			father = scope:father
			dynasty = inherit
		}
	}

	ai_check_interval = 0
}
