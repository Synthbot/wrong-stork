# Spawn Children
wrong_stork.0101 = {
	desc = "Spawns a child."
	immediate = {
		if = {
			limit = {
				is_female = yes
			}
			save_scope_as = mother
			scope:spouse = {
				save_scope_as = father
			}
		}
		else = {
			save_scope_as = father
			scope:spouse = {
				save_scope_as = mother
			}
		}
		create_character = {
			age = { 1 15 }
			gender_female_chance = 50
			trait = wild_oat
			mother = scope:mother
			father = scope:father
			dynasty = inherit
		}
	}
}
